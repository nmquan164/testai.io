        // Function to create a card for a province

        function createProvinceCard(province) {

        // Function to create a Chart.js pie chart

        function createChart(canvas, data) {

            const chartData = {

                labels: ['Nông-Lâm-Thủy sản', 'Công nghiệp-Xây dựng', 'Dịch vụ'],

                datasets: [{

                    label: 'Cơ cấu kinh tế',

                    data: [data.nlts, data.cnxd, data.dv],

                    backgroundColor: [colors.nlts, colors.cnxd, colors.dv],

                    borderColor: [borderColors.nlts, borderColors.cnxd, borderColors.dv],

                    borderWidth: 1,

                    hoverOffset: 8

                }]

            };


            new Chart(canvas, {

                type: 'doughnut',

                data: chartData,

                options: {

                    responsive: true,

                    maintainAspectRatio: false,

                    plugins: {

                        legend: {

                            display: false // We use custom legend

                        },

                        tooltip: {

                            callbacks: {

                                label: function(context) {

                                    let label = context.label || '';

                                    if (label) {

                                        label += ': ';

                                    }

                                    if (context.parsed !== null) {

                                        label += context.parsed + '%';

                                    }

                                    return label;

                                }

                            }

                        }

                    },

                    cutout: '60%'

                }

            });

        }

        // Function to filter and render provinces

        function filterAndRender() {

            const searchTerm = searchInput.value.toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, "").replace(/đ/g, "d");

            const selectedRegion = regionFilter.value;


            const filteredProvinces = provinceData.filter(province => {

                const normalizedName = province.name.toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, "").replace(/đ/g, "d");

                const nameMatch = normalizedName.includes(searchTerm);

                const regionMatch = selectedRegion === 'all' || province.region === selectedRegion;

                return nameMatch && regionMatch;

            });

            

            renderProvinces(filteredProvinces);

            

            if (filteredProvinces.length === 0) {

                noResults.classList.remove('hidden');

            } else {

                noResults.classList.add('hidden');

            }

        }


        // Initial render

        renderProvinces(provinceData);


        // Event Listeners

        searchInput.addEventListener('input', filterAndRender);

        regionFilter.addEventListener('change', filterAndRender);
